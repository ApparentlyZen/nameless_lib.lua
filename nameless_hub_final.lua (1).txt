local function getHRP(model)
    return model and model:FindFirstChild("HumanoidRootPart")
end

local function isAllyWithMe(targetPlayer)
    local myGui = player:FindFirstChild("PlayerGui")
    if not myGui then return false end
    local scrolling = myGui:FindFirstChild("Main")
        and myGui.Main:FindFirstChild("Allies")
        and myGui.Main.Allies:FindFirstChild("Container")
        and myGui.Main.Allies.Container:FindFirstChild("Allies")
        and myGui.Main.Allies.Container.Allies:FindFirstChild("ScrollingFrame")
    if scrolling then
        for _, frame in pairs(scrolling:GetDescendants()) do
            if frame:IsA("ImageButton") and frame.Name == targetPlayer.Name then
                return true
            end
        end
    end
    return false
end

local function isEnemy(targetPlayer)
    if not targetPlayer or targetPlayer == player then return false end
    local myTeam = player.Team
    local targetTeam = targetPlayer.Team
    if myTeam and targetTeam then
        if myTeam.Name == "Pirates" and targetTeam.Name == "Marines" then return true end
        if myTeam.Name == "Marines" and targetTeam.Name == "Pirates" then return true end
        if myTeam.Name == "Pirates" and targetTeam.Name == "Pirates" then
            return not isAllyWithMe(targetPlayer)
        end
        if myTeam.Name == "Marines" and targetTeam.Name == "Marines" then return false end
    end
    return true
end

local function getClosestPlayer(lpHRP)
    if not lpHRP then return nil end
    if SelectedPlayer and Players[SelectedPlayer] and Players[SelectedPlayer].Character then
        local hrp = getHRP(Players[SelectedPlayer].Character)
        if hrp then return Players[SelectedPlayer] end
    end
    local closest, closestDist = nil, MaxRange
    for _, pl in ipairs(Players:GetPlayers()) do
        if pl ~= player and isEnemy(pl) and pl.Character then
            local hum = pl.Character:FindFirstChildOfClass("Humanoid")
            local hrp = getHRP(pl.Character)
            if hum and hum.Health > 0 and hrp then
                local dist = (hrp.Position - lpHRP.Position).Magnitude
                if dist < closestDist then
                    closestDist = dist
                    closest = pl
                end
            end
        end
    end
    return closest
end

local function getClosestNPC(lpHRP)
    if not lpHRP then return nil end
    local enemiesFolder = workspace:FindFirstChild("Enemies")
    if not enemiesFolder then return nil end
    local closest, closestDist = nil, MaxRange
    for _, npc in ipairs(enemiesFolder:GetChildren()) do
        if npc:IsA("Model") then
            local hum = npc:FindFirstChildOfClass("Humanoid")
            local hrp = getHRP(npc)
            if hum and hum.Health > 0 and hrp then
                local dist = (hrp.Position - lpHRP.Position).Magnitude
                if dist < closestDist then
                    closestDist = dist
                    closest = npc
                end
            end
        end
    end
    return closest
end

local function getPredictedPosition(hrp)
    if not hrp then return nil end
    if not PredictionEnabled then return hrp.Position end
    local hum = hrp.Parent:FindFirstChildOfClass("Humanoid")
    if hum and hum.WalkSpeed >= 5 then
        return hrp.Position + (hrp.Velocity * PredictionAmount)
    end
    return hrp.Position
end

local function clearHighlight()
    if currentHighlight then
        currentHighlight:Destroy()
        currentHighlight = nil
    end
end

local function applyHighlight(targetModel)
    if not HighlightEnabled or not targetModel then return end
    if currentHighlight and currentHighlight.Adornee == targetModel then return end
    clearHighlight()
    local hl = Instance.new("Highlight")
    hl.FillColor = Color3.fromRGB(255, 255, 0)
    hl.OutlineColor = Color3.fromRGB(255, 255, 0)
    hl.FillTransparency = 0.5
    hl.OutlineTransparency = 0
    hl.Adornee = targetModel
    hl.Parent = targetModel
    currentHighlight = hl
end

local function pauseSilentAim()
    if not VSkillPauseEnabled then return end
    -- Sauvegarder les états actuels
    vSkillActive = true
    -- Désactiver temporairement silent aim et aimlock
    SilentAimPlayers = false
    SilentAimNPCs = false
    AimlockPlayers = false
    AimlockNPCs = false
    if renderConnection then
        renderConnection:Disconnect()
        renderConnection = nil
    end
    if espConnection then
        espConnection:Disconnect()
        espConnection = nil
    end
end

local function resumeSilentAim()
    if not VSkillPauseEnabled then return end
    vSkillActive = false
    -- Restaurer les états d'origine
    SilentAimPlayers = UserWantsPlayerAim  -- ces variables seront définies dans l'UI
    SilentAimNPCs = UserWantsNPCAim
    AimlockPlayers = UserWantsAimlockPlayers
    AimlockNPCs = UserWantsAimlockNPCs
    -- Redémarrer les boucles si nécessaire
    if SilentAimPlayers or SilentAimNPCs then
        if renderConnection then
            renderConnection:Disconnect()
            renderConnection = nil
        end
        renderConnection = RunService.RenderStepped:Connect(function()
            -- (même code que plus bas)
        end)
    end
    if ESPEnabled then
        if espConnection then
            espConnection:Disconnect()
            espConnection = nil
        end
        espConnection = RunService.RenderStepped:Connect(updateESP)
    end
end

-- Fonction pour gérer le temps de pause (durée approximative du skill)
local function startVSkillPause(duration)
    if vSkillTimer then task.cancel(vSkillTimer) end
    pauseSilentAim()
    vSkillTimer = task.delay(duration, function()
        resumeSilentAim()
        vSkillTimer = nil
    end)
end

-- =====================================================================
-- RENDER LOOP SILENT AIM (redirection des projectiles)
-- =====================================================================
local function startRenderLoop()
    if renderConnection then return end
    renderConnection = RunService.RenderStepped:Connect(function()
        if vSkillActive then return end  -- Ne rien faire si un VSkill est actif
        local char = player.Character
        if not char then return end
        local lpHRP = char:FindFirstChild("HumanoidRootPart")
        if not lpHRP then return end
        if not SilentAimPlayers and not SilentAimNPCs then
            PlayersPosition = nil
            NPCPosition = nil
            clearHighlight()
            return
        end

        local targetModel = nil
        if SilentAimPlayers then
            local targetPlayer = getClosestPlayer(lpHRP)
            if targetPlayer and targetPlayer.Character then
                local hrp = getHRP(targetPlayer.Character)
                PlayersPosition = getPredictedPosition(hrp)
                targetModel = targetPlayer.Character
                applyHighlight(targetModel)
            else
                PlayersPosition = nil
            end
        else
            PlayersPosition = nil
        end

        if SilentAimNPCs then
            local npc = getClosestNPC(lpHRP)
            if npc then
                local hrp = getHRP(npc)
                NPCPosition = getPredictedPosition(hrp)
                if not targetModel then
                    targetModel = npc
                    applyHighlight(targetModel)
                end
            else
                NPCPosition = nil
            end
        else
            NPCPosition = nil
        end

        -- Modification du Mouse.Hit pour rediriger les projectiles
        if MouseModule and typeof(MouseModule) == "Instance" then
            local ok, MouseData = pcall(require, MouseModule)
            if ok and type(MouseData) == "table" then
                local targetPos = PlayersPosition or NPCPosition
                if targetPos then
                    pcall(function()
                        MouseData.Hit = CFrame.new(targetPos)
                        MouseData.Target = nil
                    end)
                end
            end
        end
    end)
end

local function stopRenderLoop()
    if renderConnection then
        renderConnection:Disconnect()
        renderConnection = nil
    end
    PlayersPosition = nil
    NPCPosition = nil
    clearHighlight()
end

local function updateTargetUI()
    if not TargetInfo then
        if targetUI then targetUI.Enabled = false end
        return
    end
    if not targetUI then
        targetUI, targetNameLabel, targetHpFill = createTargetUI()
    else
        targetUI.Enabled = true
    end

    local targetChar = nearestZTarget
    if targetChar and targetChar:FindFirstChild("Humanoid") then
        local hum = targetChar.Humanoid
        targetNameLabel.Text = targetChar.Name
        local fill = math.clamp(hum.Health / hum.MaxHealth, 0, 1)
        targetHpFill.Size = UDim2.new(fill, 0, 1, 0)
        targetHpFill.BackgroundColor3 = Color3.fromRGB(255 * (1 - fill), 255 * fill, 0)
    else
        targetNameLabel.Text = "No target"
        targetHpFill.Size = UDim2.new(1, 0, 1, 0)
        targetHpFill.BackgroundColor3 = Color3.fromRGB(0, 255, 0)
    end
end

local function stopAimlockZ()
    if not aimlockActive then return end
    aimlockActive = false
    nearestZTarget = nil
    if aimRenderConn then aimRenderConn:Disconnect() aimRenderConn = nil end
    if aimTimeoutTask then task.cancel(aimTimeoutTask) aimTimeoutTask = nil end
end

local function startAimlockZ()
    if not ZSkillsEnabled or aimlockActive then return end
    nearestZTarget = getNearestZTarget()
    if not nearestZTarget then return end
    aimlockActive = true
    aimRenderConn = RunService.RenderStepped:Connect(function()
        if not ZSkillsEnabled then
            stopAimlockZ()
            return
        end
        if aimlockActive and nearestZTarget and nearestZTarget:FindFirstChild("HumanoidRootPart") then
            local cam = workspace.CurrentCamera
            if cam then
                cam.CFrame = CFrame.lookAt(cam.CFrame.Position, nearestZTarget.HumanoidRootPart.Position)
            end
        else
            stopAimlockZ()
        end
    end)
    aimTimeoutTask = task.delay(1, function()
        if aimlockActive then stopAimlockZ() end
    end)
end

local oldNamecall = hookmetamethod(game, "__namecall", function(self, ...)
    local method = getnamecallmethod()
    local args = {...}
    if method == "InvokeServer" or method == "FireServer" then
        local a1 = args[1]
        -- Détection des skills pour VSkill
        if typeof(a1) == "string" and VSkillPauseEnabled then
            if currentTool then
                if currentTool.Name == "Shark Anchor" and a1:upper() == "Z" then
                    startVSkillPause(1.5)  -- durée approximative du skill
                elseif currentTool.Name == "Cursed Dual Katana" and a1:upper() == "Z" then
                    startVSkillPause(1.2)
                elseif currentTool.Name == "Dough-Dough" and a1:upper() == "V" then
                    startVSkillPause(1.8)
                end
            end
        end
        -- Détection du Z de Godhuman pour ZSkill
        if typeof(a1) == "string" and a1:upper() == "Z" and currentZTool and currentZTool.Name == "Godhuman" then
            godhumanZActive = true
            task.spawn(function()
                task.wait(0.05)
                if ZSkillsEnabled then startAimlockZ() end
            end)
        end
    end
    return oldNamecall(self, ...)
end)

-- Hook pour le silent aim unifié (redirection des Vector3)
local hookSilent = hookmetamethod(game, "__namecall", function(self, ...)
    local method = getnamecallmethod():lower()
    local args = {...}
    if method == "fireserver" or method == "invokeserver" then
        if vSkillActive then return oldNamecall(self, ...) end  -- Ne pas rediriger si VSkill actif
        local targetPos = nil
        if SilentAimPlayers and PlayersPosition then
            targetPos = PlayersPosition
        elseif SilentAimNPCs and NPCPosition then
            targetPos = NPCPosition
        end
        if targetPos then
            if typeof(args[1]) == "Vector3" then
                args[1] = targetPos
                return oldNamecall(self, unpack(args))
            end
            if typeof(args[2]) == "Vector3" then
                args[2] = targetPos
                return oldNamecall(self, unpack(args))
            end
        end
    end
    return oldNamecall(self, ...)
end)

local function updateESP()
    if not ESPEnabled then
        for _, obj in pairs(espObjects) do pcall(function() obj:Remove() end) end
        espObjects = {}
        return
    end
    local lpChar = player.Character
    if not lpChar then return end
    local lpHRP = lpChar:FindFirstChild("HumanoidRootPart")
    if not lpHRP then return end

    for _, pl in ipairs(Players:GetPlayers()) do
        if pl ~= player and isEnemy(pl) and pl.Character then
            local char = pl.Character
            local hrp = char:FindFirstChild("HumanoidRootPart")
            local hum = char:FindFirstChildOfClass("Humanoid")
            if hrp and hum and hum.Health > 0 then
                local dist = (hrp.Position - lpHRP.Position).Magnitude
                local screenPos, onScreen = camera:WorldToViewportPoint(hrp.Position)
                if onScreen then
                    local key = pl.UserId
                    local obj = espObjects[key]
                    if not obj then
                        obj = {
                            name = Drawing.new("Text"),
                            dist = Drawing.new("Text"),
                            box = Drawing.new("Quad"),
                            health = Drawing.new("Line")
                        }
                        obj.name.Outline = true
                        obj.name.Font = 2
                        obj.name.Size = 12
                        obj.dist.Outline = true
                        obj.dist.Font = 2
                        obj.dist.Size = 10
                        obj.box.Thickness = 1
                        obj.health.Thickness = 3
                        espObjects[key] = obj
                    end
                    local sizeY = 4
                    local sizeX = 2
                    local top = Vector2.new(screenPos.X, screenPos.Y - sizeY)
                    local bottom = Vector2.new(screenPos.X, screenPos.Y + sizeY)
                    obj.name.Text = pl.Name
                    obj.name.Position = Vector2.new(screenPos.X, top.Y - 15)
                    obj.dist.Text = dist .. "m"
                    obj.dist.Position = Vector2.new(screenPos.X, bottom.Y + 5)
                    obj.box.PointA = Vector2.new(screenPos.X - sizeX, top.Y)
                    obj.box.PointB = Vector2.new(screenPos.X + sizeX, top.Y)
                    obj.box.PointC = Vector2.new(screenPos.X + sizeX, bottom.Y)
                    obj.box.PointD = Vector2.new(screenPos.X - sizeX, bottom.Y)
                    obj.box.Color = Color3.fromRGB(255, 0, 0)
                    local healthPercent = hum.Health / hum.MaxHealth
                    local barX = screenPos.X + sizeX + 5
                    local barTop = top.Y
                    local barBottom = bottom.Y
                    obj.health.From = Vector2.new(barX, barTop + (1 - healthPercent) * (barBottom - barTop))
                    obj.health.To = Vector2.new(barX, barBottom)
                    obj.health.Color = Color3.fromRGB(0, 255, 0)
                    obj.health.Visible = ESPHealthBar
                    obj.name.Visible = ESPName
                    obj.dist.Visible = ESPDistance
                    obj.box.Visible = ESPBox
                else
                    local obj = espObjects[pl.UserId]
                    if obj then
                        obj.name.Visible = false
                        obj.dist.Visible = false
                        obj.box.Visible = false
                        obj.health.Visible = false
                    end
                end
            end
        end
    end
end

-- Auto Ken
local function startAutoKen()
    if autoKenLoop then autoKenLoop:Disconnect() end
    autoKenLoop = RunService.Heartbeat:Connect(function()
        if not AutoKen then
            if autoKenLoop then autoKenLoop:Disconnect() autoKenLoop = nil end
            return
        end
        local char = player.Character
        if not char then return end
        local kenButton = player.PlayerGui:FindFirstChild("MobileContextButtons") and
                          player.PlayerGui.MobileContextButtons.ContextButtonFrame and
                          player.PlayerGui.MobileContextButtons.ContextButtonFrame:FindFirstChild("BoundActionKen")
        if kenButton and kenButton:GetAttribute("Selected") ~= true then
            kenButton:SetAttribute("Selected", true)
        end
        local commE = RS:FindFirstChild("Remotes") and RS.Remotes:FindFirstChild("CommE")
        if commE then
            pcall(function() commE:FireServer("Ken", true) end)
        end
    end)
end
if AutoKen then startAutoKen() end

-- Fly (simple)
local function startFly()
    if flyConn then flyConn:Disconnect() end
    local char = player.Character
    if not char then return end
    local hrp = char:FindFirstChild("HumanoidRootPart")
    if not hrp then return end
    local hum = char:FindFirstChildOfClass("Humanoid")
    if hum then hum.PlatformStand = true end
    local att = Instance.new("Attachment", hrp)
    local lv = Instance.new("LinearVelocity", hrp)
    lv.MaxForce = math.huge
    lv.VelocityConstraintMode = Enum.VelocityConstraintMode.Vector
    lv.Attachment0 = att
    flyConn = RunService.RenderStepped:Connect(function()
        if not FlyEnabled then
            if lv then lv:Destroy() end
            if att then att:Destroy() end
            if hum then hum.PlatformStand = false end
            if flyConn then flyConn:Disconnect() flyConn = nil end
            return
        end
        local moveDir = Vector3.new(0, 0, 0)
        local cf = camera.CFrame
        if UserInputService:IsKeyDown(Enum.KeyCode.W) then moveDir = moveDir + cf.LookVector end
        if UserInputService:IsKeyDown(Enum.KeyCode.S) then moveDir = moveDir - cf.LookVector end
        if UserInputService:IsKeyDown(Enum.KeyCode.A) then moveDir = moveDir - cf.RightVector end
        if UserInputService:IsKeyDown(Enum.KeyCode.D) then moveDir = moveDir + cf.RightVector end
        if UserInputService:IsKeyDown(Enum.KeyCode.Space) then moveDir = moveDir + Vector3.new(0, 1, 0) end
        if UserInputService:IsKeyDown(Enum.KeyCode.LeftControl) then moveDir = moveDir - Vector3.new(0, 1, 0) end
        lv.VectorVelocity = moveDir.Magnitude > 0 and moveDir.Unit * FlySpeed or Vector3.new(0, 0, 0)
    end)
end
if FlyEnabled then startFly() end
